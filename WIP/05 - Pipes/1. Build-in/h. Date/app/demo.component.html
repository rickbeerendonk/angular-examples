{{ date | date }} = Default

{{ date | date: 'full' }} = Format: 'full'

{{ date | date: 'mm ss' }} = Format: 'mm ss'

{{ date | date: 'short':'+0430' }} = Timezone +0430 (short format)

{{ date | date: 'full':'':'en-US' }} = English

{{ date | date: 'full':'':'nl' }} = Dutch

{{ date | date: 'full':'':'el' }} = Greek
